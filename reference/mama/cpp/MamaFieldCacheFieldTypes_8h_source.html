396 template <typename T>
397 class MamaFieldCacheFieldVector
398 {
399 protected:
400  MamaFieldCacheFieldVector()
401  : mValues(NULL)
402  , mSize(0)
403  {
404  }
405  ~MamaFieldCacheFieldVector()
406  {
407  if (!mValues || !mSize)
408  {
409  return;
410  }
411  delete[] mValues;
412  }
413 
414  void grow(mama_size_t newSize) const
415  {
416  if (newSize <= mSize)
417  return;
418 
419  T* newVector = new T[newSize];
420  for (mama_size_t i = 0; i < mSize; ++i)
421  {
422  newVector[i] = mValues[i];
423  }
424 
425  if (mValues)
426  delete[] mValues;
427 
428  mValues = newVector;
429  mSize = newSize;
430  }
431 
432 protected:
433  mutable T* mValues;
434  mutable mama_size_t mSize;
435 };
